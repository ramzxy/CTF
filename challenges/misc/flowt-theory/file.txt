<script>
const rawReceipts = {"Alice":100,"Bob":10};

function addRow() {
    const container = document.getElementById('inputs-container');
    const div = document.createElement('div');
    div.className = 'input-row';
    div.innerHTML = `
        <input type="text" name="names[]" placeholder="Filename">
        <input type="text" name="amounts[]" placeholder="0.00">
    `;
    container.appendChild(div);
}

function calculateDebts() {
    const peopleNames = Object.keys(rawReceipts);
    if (peopleNames.length < 2) return; 

    let people = [];
    let total = 0;

    peopleNames.forEach(name => {
        let paid = parseFloat(rawReceipts[name]);
        if (!isNaN(paid)) {
            total += paid;
            people.push({ name: name, paid: paid, balance: 0 });
        }
    });
    
    if (total === 0) return;

    let average = total / people.length;
    
    people.forEach(p => {
        p.balance = p.paid - average;
    });

    let debtors = people.filter(p => p.balance < -0.01).sort((a, b) => a.balance - b.balance);
    let creditors = people.filter(p => p.balance > 0.01).sort((a, b) => b.balance - a.balance);

    const container = document.getElementById('settlement-list');
    const card = document.getElementById('settlement-card');
    container.innerHTML = ''; 

    if (debtors.length > 0 && creditors.length > 0) {
        card.style.display = 'block';

        let i = 0; 
        let j = 0; 

        while (i < debtors.length && j < creditors.length) {
            let debtor = debtors[i];
            let creditor = creditors[j];
            let amount = Math.min(Math.abs(debtor.balance), creditor.balance);
            amount = Math.round(amount * 100) / 100;

            if (amount > 0) {
                let div = document.createElement('div');
                div.className = 'settlement-row';
                div.innerHTML = `
                    <span>${debtor.name}</span>
                    <span class="arrow">pays</span>
                    <span>${creditor.name}</span>
                    <span style="flex-grow:1"></span>
                    <span class="money">$${amount.toFixed(2)}</span>
                `;
                container.appendChild(div);
            }

            debtor.balance += amount;
            creditor.balance -= amount;

            if (Math.abs(debtor.balance) < 0.01) i++;
            if (creditor.balance < 0.01) j++;
        }
    }
}
window.onload = calculateDebts;
</script>
